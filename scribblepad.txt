namespace NaraiNirai.AgenticAI.Tests.Integration.Tests.Execution
{
    using NaraiNirai.AgenticAI.Core.Enums;
    using NaraiNirai.AgenticAI.Execution.Builders;
    using NaraiNirai.AgenticAI.Execution.Extensions;
    using NaraiNirai.AgenticAI.Execution.Models;
    using NaraiNirai.AgenticAI.Execution.Workflow;
    using Xunit;

    public class ExecutionEngineTests : TestBase
    {
        [Fact]
        public async Task WorkflowExecutor_ExecutesSequentialSteps()
        {
            // Arrange
            var workflowExecutor = ServiceProvider.GetRequiredService<WorkflowExecutor>();

            var workflow = new WorkflowBuilder()
                .AddStep("Step1", "First step", ToolType.None, "", "Step2")
                .AddStep("Step2", "Second step", ToolType.None, "", null)
                .Build();

            var context = new TestExecutionContext().AsWorkflowContext();
            context.WorkflowState["Test"] = "Initial";

            // Act
            var result = await workflowExecutor.ExecuteAsync(workflow, context, CancellationToken.None);

            // Assert
            Assert.True(result.IsSuccess);
            Assert.Equal(2, result.Outputs["ExecutedSteps"]);
        }

        [Fact]
        public async Task WorkflowExecutor_HandlesConditionalBranching()
        {
            // Arrange
            var workflowExecutor = ServiceProvider.GetRequiredService<WorkflowExecutor>();

            var workflow = new WorkflowBuilder()
                .AddStep("Start", "Start", ToolType.None, "", "Condition")
                .AddStep("Condition", "Check condition", ToolType.None, "", "TruePath")
                    .WithCondition("WorkflowState['Value'] > 5")
                .AddStep("TruePath", "True path", ToolType.None, "", null)
                .AddStep("FalsePath", "False path", ToolType.None, "", null)
                .Build();

            // Test true condition
            var contextTrue = new TestExecutionContext().AsWorkflowContext();
            contextTrue.WorkflowState["Value"] = 10;

            var resultTrue = await workflowExecutor.ExecuteAsync(workflow, contextTrue, CancellationToken.None);
            Assert.True(resultTrue.IsSuccess);

            // Test false condition
            var contextFalse = new TestExecutionContext().AsWorkflowContext();
            contextFalse.WorkflowState["Value"] = 3;

            var resultFalse = await workflowExecutor.ExecuteAsync(workflow, contextFalse, CancellationToken.None);
            Assert.True(resultFalse.IsSuccess);
        }

        [Fact]
        public async Task WorkflowExecutor_HandlesLoops()
        {
            // Arrange
            var workflowExecutor = ServiceProvider.GetRequiredService<WorkflowExecutor>();

            var workflow = new WorkflowBuilder()
                .AddStep("Initialize", "Initialize", ToolType.None, "", "LoopStart")
                .AddStep("LoopStart", "Loop start", ToolType.None, "", "Increment")
                    .AsLoopStart()
                .AddStep("Increment", "Increment", ToolType.None, "", "LoopEnd")
                .AddStep("LoopEnd", "Loop end", ToolType.None, "", "LoopStart")
                    .AsLoopEnd("iteration < 3")
                .Build();

            var context = new TestExecutionContext().AsWorkflowContext();

            // Act
            var result = await workflowExecutor.ExecuteAsync(workflow, context, CancellationToken.None);

            // Assert
            Assert.True(result.IsSuccess);
            Assert.Equal(3, result.Outputs["ExecutedSteps"]); // Loop runs 3 times
        }

        private sealed class TestExecutionContext : IExecutionContext
        {
            public string Id => Guid.NewGuid().ToString("N");
            public string SessionId => "test-session";
            public string UserId => "test-user";
            public DateTimeOffset CreatedAt => DateTimeOffset.UtcNow;
            public DateTimeOffset? ExpiresAt => null;
            public IDictionary<string, object> Metadata { get; } = new Dictionary<string, object>();
            public IReadOnlyCollection<IExecutionStep> Steps { get; } = Array.Empty<IExecutionStep>();
            public CancellationToken CancellationToken => CancellationToken.None;
        }

namespace NaraiNirai.AgenticAI.Tests.Integration.Tests.Tools
{
    using NaraiNirai.AgenticAI.Core.Contracts;
    using NaraiNirai.AgenticAI.Tools.Basic.FileTools;
    using System.Text.Json;
    using Xunit;

    public class FileToolsTests : TestBase
    {
        private readonly string _testFilesDir = Path.Combine(Directory.GetCurrentDirectory(), "TestFiles");

        public FileToolsTests()
        {
            Directory.CreateDirectory(_testFilesDir);
        }

        [Fact]
        public async Task JsonTool_ReadsValidJson()
        {
            // Arrange
            var jsonFile = Path.Combine(_testFilesDir, "test.json");
            var testData = new { Name = "Test", Value = 123 };
            await File.WriteAllTextAsync(jsonFile, JsonSerializer.Serialize(testData));

            var jsonTool = new JsonTool();
            var context = new TestExecutionContext();

            // Act
            var result = await jsonTool.ExecuteAsync(context, $"read|{jsonFile}", CancellationToken.None);

            // Assert
            Assert.True(result.IsSuccess);
            Assert.Contains("Content", result.Outputs.Keys);
            Assert.Contains("Test", result.Outputs["Content"].ToString());
        }

        [Fact]
        public async Task JsonTool_WritesValidJson()
        {
            // Arrange
            var jsonFile = Path.Combine(_testFilesDir, "output.json");
            var jsonTool = new JsonTool();
            var context = new TestExecutionContext();
            var testData = JsonSerializer.Serialize(new { Key = "Value" });

            // Act
            var result = await jsonTool.ExecuteAsync(context, $"write|{jsonFile}|{testData}", CancellationToken.None);

            // Assert
            Assert.True(result.IsSuccess);
            Assert.True(File.Exists(jsonFile));
            var content = await File.ReadAllTextAsync(jsonFile);
            Assert.Contains("Key", content);
        }

        [Fact]
        public async Task JsonTool_FailsForInvalidJson()
        {
            // Arrange
            var jsonFile = Path.Combine(_testFilesDir, "invalid.json");
            await File.WriteAllTextAsync(jsonFile, "{ invalid json }");

            var jsonTool = new JsonTool();
            var context = new TestExecutionContext();

            // Act
            var result = await jsonTool.ExecuteAsync(context, $"read|{jsonFile}", CancellationToken.None);

            // Assert
            Assert.False(result.IsSuccess);
            Assert.Contains("not valid JSON", result.ErrorMessage);
        }

        [Fact]
        public async Task TxtTool_ReadsAndWritesText()
        {
            // Arrange
            var txtFile = Path.Combine(_testFilesDir, "test.txt");
            var testContent = "Hello, ValaiORS!";
            await File.WriteAllTextAsync(txtFile, testContent);

            var txtTool = new TxtTool();
            var context = new TestExecutionContext();

            // Act: Read
            var readResult = await txtTool.ExecuteAsync(context, $"read|{txtFile}", CancellationToken.None);
            Assert.True(readResult.IsSuccess);
            Assert.Equal(testContent, readResult.Outputs["Content"]);

            // Act: Write
            var newContent = "Updated content";
            var writeResult = await txtTool.ExecuteAsync(context, $"write|{txtFile}|{newContent}", CancellationToken.None);
            Assert.True(writeResult.IsSuccess);

            // Verify
            var updatedContent = await File.ReadAllTextAsync(txtFile);
            Assert.Equal(newContent, updatedContent);
        }

        {

    using NaraiNirai.AgenticAI.Retry.Strategies;
    using Xunit;

    public class RetryTests : TestBase
    {
        [Fact]
        public async Task MaxAttemptsRetryPolicy_AllowsRetry_WithinLimit()
        {
            // Arrange
            var policy = new MaxAttemptsRetryPolicy(3);
            var context = new TestExecutionContext();
            var result = new ExecutionResult("test-id", "test-context", false);

            // Act & Assert
            Assert.True(await policy.ShouldRetryAsync(context, result, 0));
            Assert.True(await policy.ShouldRetryAsync(context, result, 1));
            Assert.True(await policy.ShouldRetryAsync(context, result, 2));
            Assert.False(await policy.ShouldRetryAsync(context, result, 3)); // Exceeds max attempts
        }

        [Fact]
        public async Task RetryStrategy_ExecutesWithRetry()
        {
            // Arrange
            var policy = new MaxAttemptsRetryPolicy(3);
            var strategy = new RetryStrategy(policy);

            var attemptCount = 0;
            var context = new TestExecutionContext();

            // Act
            var result = await strategy.ExecuteWithRetryAsync(
                async (ctx, ct) =>
                {
                    attemptCount++;
                    if (attemptCount < 3)
                    {
                        return new ExecutionResult("test-id", "test-context", false, "Temporary failure");
                    }
                    return new ExecutionResult("test-id", "test-context", true);
                },
                context,
                CancellationToken.None);

            // Assert
            Assert.True(result.IsSuccess);
            Assert.Equal(3, attemptCount);
        }

        [Fact]
        public async Task CircuitBreakerRetryPolicy_OpensAfterThreshold()
        {
            // Arrange
            var policy = new CircuitBreakerRetryPolicy(failureThreshold: 2, resetTimeout: TimeSpan.FromSeconds(10));
            var context = new TestExecutionContext();
            var result = new ExecutionResult("test-id", "test-context", false);

            // Act: First failure
            Assert.True(await policy.ShouldRetryAsync(context, result, 0));

            // Act: Second failure (threshold reached)
            Assert.True(await policy.ShouldRetryAsync(context, result, 1));

            // Act: Third failure (circuit should be open)
            Assert.False(await policy.ShouldRetryAsync(context, result, 2));
        }
